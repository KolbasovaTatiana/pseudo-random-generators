60class LCG_BaysDurham {
61 static const uint64_t A = 6364136223846793005ULL;
62 static const uint64_t C = 1442695040888963407ULL;
63 static const int table_size = 32;
64
65 uint64_t state;
66 array<uint64_t, table_size> table{};
67 uint64_t last;
68
71 uint64_t rawLCG() {
72 state = A * state + C;
73 return state;
74 }
75
76public:
79 explicit LCG_BaysDurham(uint64_t seed = 123456789ULL) {
80 if (seed != 0)
81 state = seed;
82 else
83 state = 123456789ULL;
84 last = 0;
85 for (int i = 0; i < table_size; ++i) table[i] = rawLCG();
86 last = rawLCG();
87 }
88
91 void seed(uint64_t seedValue) {
92 state = seedValue ? seedValue : 123456789ULL;
93 for (int i = 0; i < table_size; ++i) table[i] = rawLCG();
94 last = rawLCG();
95 }
96
99 uint64_t next64() {
100 uint64_t x = rawLCG();
101 size_t j = static_cast<size_t>((last >> 59) & (table_size - 1));
102 uint64_t y = table[j];
103 table[j] = x;
104 last = y;
105
106 y ^= (y >> 27);
107 y *= 0x3C79AC492BA7B653ULL;
108 y ^= (y >> 33);
109 y *= 0x1C69B3F74AC4AE35ULL;
110 y ^= (y >> 27);
111 return y;
112 }
113
116 uint32_t next() {
117 return static_cast<uint32_t>(next64());
118 }

176class AdditiveLFG_RotateXor {
177 static const int LAG_A = 55;
178 static const int LAG_B = 24;
179
180 array<uint32_t, LAG_A> buffer{};
181 int index;
182
186 static uint32_t scramble(uint32_t x) {
187 x = rotl32(x, 7) ^ (x >> 3);
188 x += 0x9e3779b9U;
189 x ^= rotl32(x, 11);
190 x ^= (x >> 16);
191 return x;
192 }
193
195 void refill() {
196 for (int i = 0; i < LAG_A; ++i) {
197 int j = (i + LAG_B) % LAG_A;
198 uint32_t v = buffer[i] + buffer[j];
199 v ^= rotl32(buffer[j], 9);
200 v = scramble(v);
201 buffer[i] = v;
202 }
203 index = 0;
204 }
205
206public:
209 explicit AdditiveLFG_RotateXor(uint32_t seed = 123456789U) {
210 this->seed(seed);
211 }
212
215 void seed(uint32_t seedValue) {
216 uint32_t x = seedValue ? seedValue : 123456789U;
217 for (int i = 0; i < LAG_A; ++i) {
218 x = 1664525U * x + 1013904223U;
219 buffer[i] = scramble(x + static_cast<uint32_t>(i * 2654435761U));
220 }
221 index = LAG_A;
222 for (int k = 0; k < 8; ++k) refill();
223 }
224
227 uint32_t next32() {
228 if (index >= LAG_A) refill();
229 return buffer[index++];
230 }
231
234 uint32_t next() {
235 return next32();
236 }

251SampleStats computeStats(const vector<int>& sample) {
252 int n = static_cast<int>(sample.size());
253 double sum = 0.0;
254 for (int x : sample) sum += x;
255 double mean = sum / n;
256
257 double sq = 0.0;
258 for (int x : sample) sq += (x - mean) * (x - mean);
259 double stddev = sqrt(sq / (n - 1));
260 double cv = (mean != 0.0) ? (stddev / mean) : 0.0;
261
262 int bins[K] = { 0 };
263 for (int x : sample) {
264 int bin = x / BIN_SIZE;
265 if (bin >= K) bin = K - 1;
266 ++bins[bin];
267 }
268
269 double expected = double(n) / K;
270 double chi2 = 0.0;
271 for (int i = 0; i < K; ++i) {
272 double diff = bins[i] - expected;
273 chi2 += (diff * diff) / expected;
274 }
275
276 bool uniform = (chi2 <= CHI2_CRIT);
277 return { mean, stddev, cv, chi2, uniform };
278}

298double my_erfc(double x) {
299 double t = 1.0 / (1.0 + 0.5 * fabs(x));
300 double tau = t * exp(-x * x - 1.26551223 + t * (1.00002368 + t * (0.37409196 +
301 t * (0.09678418 + t * (-0.18628806 + t * (0.27886807 + t * (-1.13520398 +
302 t * (1.48851587 + t * (-0.82215223 + t * 0.17087277)))))))));
303 return (x >= 0) ? tau : 2.0 - tau;
304}

310double p_value_from_chi2(double chi2, int df) {
311 double z = (sqrt(2.0 * chi2) - sqrt(2.0 * df - 1.0));
312 return my_erfc(fabs(z) / sqrt(2.0));
313}

318double nist_monobit(const vector<uint8_t>& bits) {
319 int n = static_cast<int>(bits.size());
320 if (n < 100) return -1;
321 int sum = 0;
322 for (auto b : bits) sum += b ? 1 : -1;
323 double s = sum / sqrt(static_cast<double>(n));
324 return my_erfc(fabs(s) / sqrt(2.0));
325}

331double nist_block_frequency(const vector<uint8_t>& bits, int M = 128) {
332 int n = static_cast<int>(bits.size());
333 if (n < M) return -1;
334 int N = n / M;
335 double chi2 = 0.0;
336 for (int i = 0; i < N; ++i) {
337 int ones = 0;
338 for (int j = 0; j < M; ++j)
339 if (bits[i * M + j]) ++ones;
340 double pi = static_cast<double>(ones) / M;
341 chi2 += (pi - 0.5) * (pi - 0.5);
342 }
343 chi2 *= 4.0 * M;
344 return p_value_from_chi2(chi2, N);
345}

350double nist_runs(const vector<uint8_t>& bits) {
351 int n = static_cast<int>(bits.size());
352 if (n < 100) return -1;
353 int ones = 0;
354 for (auto b : bits) ones += b;
355 double pi = static_cast<double>(ones) / n;
356 if (fabs(pi - 0.5) >= 2.0 / sqrt(static_cast<double>(n))) return 0.0;
357 int runs = 1;
358 for (int i = 1; i < n; ++i)
359 if (bits[i] != bits[i - 1]) ++runs;
360 double num = runs - 2.0 * n * pi * (1.0 - pi);
361 double den = 2.0 * sqrt(2.0 * n) * pi * (1.0 - pi);
362 double z = num / den;
363 return my_erfc(fabs(z) / sqrt(2.0));
364}

369double nist_longest_run(const vector<uint8_t>& bits) {
370 int n = static_cast<int>(bits.size());
371 if (n < 128) return -1;
372
373 const int M = 10000;
374 int N = n / M;
375 if (N == 0) return -1;
376 const int CAT = 7;
377 vector<int> v(CAT, 0);
378
379 for (int i = 0; i < N; ++i) {
380 int longest = 0, current = 0;
381 for (int j = 0; j < M; ++j) {
382 if (bits[i * M + j]) ++current;
383 else {
384 if (current > longest) longest = current;
385 current = 0;
386 }
387 }
388 if (current > longest) longest = current;
389
390 if (longest <= 10) v[0]++;
391 else if (longest == 11) v[1]++;
392 else if (longest == 12) v[2]++;
393 else if (longest == 13) v[3]++;
394 else if (longest == 14) v[4]++;
395 else if (longest == 15) v[5]++;
396 else v[6]++;
397 }
398
399 double pi[CAT] = { 0.0882, 0.2092, 0.2483, 0.1933, 0.1208, 0.0675, 0.0727 };
400 double chi2 = 0.0;
401 for (int i = 0; i < CAT; ++i) {
402 double expected = N * pi[i];
403 chi2 += (v[i] - expected) * (v[i] - expected) / expected;
404 }
405 return p_value_from_chi2(chi2, CAT - 1);
406}

412double nist_cusum(const vector<uint8_t>& bits, bool forward = true) {
413 int n = static_cast<int>(bits.size());
414 if (n < 100) return -1;
415 vector<int> X(n, 0);
416 int sum = 0;
417 if (forward) {
418 for (int i = 0; i < n; ++i) {
419 sum += bits[i] ? 1 : -1;
420 X[i] = sum;
421 }
422 }
423 else {
424 for (int i = n - 1; i >= 0; --i) {
425 sum += bits[i] ? 1 : -1;
426 X[n - 1 - i] = sum;
427 }
428 }
429 int maxZ = 0;
430 for (auto z : X) if (abs(z) > maxZ) maxZ = abs(z);
431 double z_stat = maxZ / sqrt(static_cast<double>(n));
432 return my_erfc(z_stat / sqrt(2.0));
433}

477template<typename Gen>
478double measureGenerationTime(Gen& gen, int count) {
479 auto t0 = chrono::high_resolution_clock::now();
480 for (int i = 0; i < count; ++i) {
481 volatile uint32_t sink = gen.next();
482 (void)sink;
483 }
484 auto t1 = chrono::high_resolution_clock::now();
485 return chrono::duration<double, milli>(t1 - t0).count();
486}
